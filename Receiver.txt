#include <SPI.h>
#include <nRF24L01.h>
#include <RF24.h>
#include <Adafruit_GFX.h>
#include <Adafruit_PCD8544.h>
RF24 radio(9, 10);
const byte address[10] = "robotistan100";
Adafruit_PCD8544 display = Adafruit_PCD8544(7, 6, 5, 4, 3);
float list[3];
void setup() {
  Serial.begin(9600);
  radio.begin();
  radio.openReadingPipe(0, address);
  radio.setChannel(0x76);
  radio.setPALevel(RF24_PA_MAX);
  radio.setDataRate(RF24_250KBPS);
  radio.startListening();
  radio.enableDynamicPayloads();
  radio.powerUp();
  display.begin();
  display.setContrast(57);
  display.clearDisplay();
}
void loop() {
  if (radio.available()) {
    radio.read(list, sizeof(list));
    int okunan = analogRead(A0);
    Serial.println(list[0]);
    Serial.println(list[1]);
    Serial.println(list[2]);
    if(okunan < 512)
    {
    display.display();
    display.clearDisplay();

    display.setTextSize(1);
    display.setTextColor(BLACK);
    display.setCursor(0,0);
    display.println("latitude: ");
    display.setTextColor(BLACK);
    display.setCursor(0,8);    
    display.println(list[0]);
    display.setTextColor(BLACK);
    display.setCursor(0,16);
    display.println("longitude: ");
    display.setTextColor(BLACK);
    display.setCursor(0,24);    
    display.println(list[1]);
    /*
    display.display();
    display.clearDisplay();
    */
    }
    else
    {
    display.display();
    display.clearDisplay();

    display.setTextSize(1);
    display.setTextColor(BLACK);
    display.setCursor(0,0);
    display.println("distance: ");
    display.setTextColor(BLACK);
    display.setCursor(0,8);    
    display.println(list[2]);  
    /*
    display.display();
    display.clearDisplay();
    */
    }
  }
  else
  {
    display.display();
    display.clearDisplay();
    display.setTextSize(1);
    display.setTextColor(BLACK);
    display.setCursor(0,0);
    display.println("Signal?");
    /*
    display.display();
    display.clearDisplay();
    */
  }
}
